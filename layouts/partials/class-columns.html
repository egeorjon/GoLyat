{{- $columns := . -}} {{- $colnum := 1 -}} {{- $colnum = (index $columns "def" ) -}} {{- $class := printf "row row-cols-%d" (int $colnum) -}} {{- $breakpoints := slice "sm" "md" "lg" "xl" -}} {{- range $brk := $breakpoints -}} {{- $cn := (index $columns $brk ) -}} {{- if ne $colnum $cn -}} {{- $class = (printf "%s row-cols-%s-%d" $class $brk (int $cn) ) -}} {{- $colnum = $cn -}} {{- end -}} {{- end -}} {{- return $class -}}
